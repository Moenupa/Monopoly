55     /**
56      * Given a string and a length, return a new string such that the original string is appended
57      * with blank spaces while the total length is equal to the given length, and the original
58      * string should be centered.
59      *
60      * @param str the original string
61      * @param length the total length of the new string
62      * @return the new string
63      */
64     public String middle(String str, int length) {
65         int len = str.length();
66         int diff = length - len;
67         int right = diff / 2;
68         int left = diff - right;
69         return " ".repeat(Math.max(0, left)) + str + " ".repeat(Math.max(0, right));
70     }
